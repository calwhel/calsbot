//@version=5
indicator(title='Trend Magic Enhanced [AlgoAlpha]', shorttitle='AlgoAlpha - Trend Magic', overlay=true, format=format.price, precision=2)

cciPeriod = input.int(20, 'CCI Period', group='Main Settings', tooltip='Period for the Commodity Channel Index calculation')
atrMultiplier = input.float(2.0, 'ATR Multiplier', group='Main Settings', tooltip='Multiplier for ATR in trend calculation')
atrPeriod = input.int(5, 'ATR Period', group='Main Settings', tooltip='Period for the Average True Range calculation')
magicTrendLength = input.int(14, 'Trend Magic Smoothing Length', tooltip='Smoothing length for Trend Magic if smoothing is enabled', group='Main Settings')
maType = input.string('SMA', title='MA Smoothing Type', options=['SMA', 'EMA', 'SMMA (RMA)', 'WMA', 'VWMA'], group='Main Settings', tooltip='Moving average type for smoothing Magic Trend')
source = input(close, 'Source', group='Main Settings', tooltip='Input source for calculations')
smoothMagicTrend = input.bool(false, 'Smooth Trend Magic', tooltip='Enable to smooth Trend Magic for trend identification', group='Main Settings')
bullishColor = input.color(#00ffbb, 'Bullish Color', group='Appearance', tooltip='Color for bullish trend')
bearishColor = input.color(#ff1100, 'Bearish Color', group='Appearance', tooltip='Color for bearish trend')

atrValue = ta.sma(ta.tr(true), atrPeriod)
trendUp = low - atrValue * atrMultiplier
trendDown = high + atrValue * atrMultiplier
cciValue = ta.cci(source, cciPeriod)

magicTrend = 0.0
if cciValue >= 0
    magicTrend := trendUp < nz(magicTrend[1]) ? nz(magicTrend[1]) : trendUp
else
    magicTrend := trendDown > nz(magicTrend[1]) ? nz(magicTrend[1]) : trendDown

movingAverage(src, length, type) =>
    switch type
        'SMA' => ta.sma(src, length)
        'EMA' => ta.ema(src, length)
        'SMMA (RMA)' => ta.rma(src, length)
        'WMA' => ta.wma(src, length)
        'VWMA' => ta.vwma(src, length)

magicTrend := smoothMagicTrend ? movingAverage(magicTrend, magicTrendLength, maType) : magicTrend

var trendDirection = 0
if ta.crossover(low, magicTrend)
    trendDirection := 1
else if ta.crossunder(high, magicTrend)
    trendDirection := -1
else
    trendDirection := nz(trendDirection[1])

trendColor = trendDirection > 0 ? color.new(bullishColor, 70) : color.new(bearishColor, 70)
magicTrendPlot = plot(magicTrend, color=trendColor, linewidth=3)
pricePlot = plot((open + close) / 2, display=display.none)
fill(magicTrendPlot, pricePlot, (open + close) / 2, magicTrend, color.new(trendDirection > 0 ? bullishColor : bearishColor, 90), color.new(trendDirection > 0 ? bullishColor : bearishColor, 70))

l1 = ta.lowest(21)
h1 = ta.highest(21)

plotshape(ta.crossover(trendDirection, 0) ? l1 : na, "Bullish Trend", shape.labelup, location.absolute, bullishColor, text = "▲", textcolor = chart.fg_color, size = size.small)
plotshape(ta.crossunder(trendDirection, 0) ? h1 : na, "Bearish Trend", shape.labeldown, location.absolute, bearishColor, text = "▼", textcolor = chart.fg_color, size = size.small)

alertcondition(trendDirection == 1 and trendDirection[1] != 1, title='Bullish Trend', message='Bullish Trend Detected')
alertcondition(trendDirection == -1 and trendDirection[1] != -1, title='Bearish Trend', message='Bearish Trend Detected')
alertcondition(trendDirection != trendDirection[1], title='Trend Shift', message='Trend Shift Detected')
